import { prisma } from "../database/database.js";
import { TipoSala as PrismaTipoSala} from "@prisma/client";

  

const criarSala = async () => {
  const sala = await prisma.sala.create({
    data: {
      nomeSala: "SalaTeste",
      tipoSala: PrismaTipoSala.COMUN, 
      disponibilidadeSala: true,
      fotoSala: "imagem.jpg",
      qtdeSala: 20,
    },
  });
};
criarSala()
  .then(() => console.log("Sala criada com sucesso!"))
  .catch((err) => console.error("Erro ao criar sala:", err));
